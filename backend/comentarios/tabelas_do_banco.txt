-- Criação do banco de dados e tabelas para o sistema de controle financeiro
-- ================================================
-- TIPOS ENUM
-- ================================================

-- ATENCAO: nao existe migracao de ALTER TYPE em main._aplicar_migracoes(). Os valores
-- 'Presidente' e 'Operador' foram acrescentados a mao nas bases existentes; este script
-- ja os cria para uma base do zero. Ao acrescentar valor a um enum, lembre das DUAS
-- pontas: o model (models/usuario.py) e o banco.
CREATE TYPE cargo_enum AS ENUM ('Presidente','Diretor','Tesoureiro','Secretário','Conselheiro','Associado','Desenvolvedor');
CREATE TYPE acesso_enum AS ENUM ('Administrador','Operador','Consulta');
CREATE TYPE natureza_enum AS ENUM ('Debito','Credito');
CREATE TYPE tipo_clifor_enum AS ENUM ('C','F','A');

CREATE EXTENSION IF NOT EXISTS pg_trgm;

-- ================================================
-- ORDEM DE CRIAÇÃO: Usuario > tipo_conta > ClienteFornecedor > Endereco > Contato > Login > Lancamento
-- ================================================

CREATE TABLE Usuario (
    id_usuario BIGINT GENERATED ALWAYS AS IDENTITY,
    Senha VARCHAR(255) NOT NULL,
    Nome VARCHAR(255) NOT NULL,
    Data_Cadastro TIMESTAMP NOT NULL DEFAULT NOW(),
    Email VARCHAR(255) NOT NULL,
    Cargo cargo_enum NOT NULL,
    Perfil_de_Acesso acesso_enum NOT NULL,
    Notificacao BOOLEAN NOT NULL DEFAULT FALSE,
    Suspenso TIMESTAMP DEFAULT NULL,
    Qtd_suspensao INT DEFAULT 0,
    Bloqueado BOOLEAN NOT NULL DEFAULT FALSE,
    Exclusao TIMESTAMP DEFAULT NULL,
    -- Nao mapeada em nenhum model e sem uso no codigo (sobrou da integracao com o
    -- Telegram; TELEGRAM_BOT_TOKEN ainda existe no config.env). Mantida porque a
    -- coluna existe nas bases atuais -- se for dropar, dropar aqui e no banco.
    telegram_id VARCHAR(255) DEFAULT NULL,
    primeiro_acesso BOOLEAN NOT NULL DEFAULT TRUE,
    PRIMARY KEY (id_usuario)
);

-- Usuario
CREATE INDEX idx_usuario_email ON Usuario(Email);
CREATE INDEX idx_usuario_bloqueado ON Usuario(Bloqueado);


CREATE TABLE tipo_conta (
    id_tipo_conta BIGINT GENERATED ALWAYS AS IDENTITY,
    Descricao_Conta VARCHAR(255) NOT NULL,
    Natureza_Conta natureza_enum NOT NULL,
    Observacao TEXT DEFAULT NULL,
    PRIMARY KEY (id_tipo_conta)
);


CREATE TABLE ClienteFornecedor (
    id_CliFor BIGINT GENERATED ALWAYS AS IDENTITY,
    id_Usuario_FK BIGINT,
    PessoaFisica_Juridica BOOLEAN NOT NULL,
    CPF_CNPJ VARCHAR(255) NOT NULL,
    RG_InscricaoEstadual VARCHAR(255) NOT NULL,
    Nome VARCHAR(255) NOT NULL,
    DataNascimento DATE NOT NULL,
    Tipo_CliFor tipo_clifor_enum NOT NULL,
    Ativo BOOLEAN NOT NULL DEFAULT TRUE,
    Inadimplente BOOLEAN NOT NULL DEFAULT FALSE,
    PRIMARY KEY (id_CliFor),
    FOREIGN KEY (id_Usuario_FK) REFERENCES Usuario(id_usuario)
);

-- ClienteFornecedor
CREATE INDEX idx_clifor_cpf_cnpj ON ClienteFornecedor(CPF_CNPJ);
CREATE INDEX idx_clifor_nome ON ClienteFornecedor(Nome);
CREATE INDEX idx_clifor_ativo ON ClienteFornecedor(Ativo);
CREATE INDEX idx_clifor_inadimplente ON ClienteFornecedor(Inadimplente);


CREATE TABLE Endereco (
    id_Endereco BIGINT GENERATED ALWAYS AS IDENTITY,
    id_CliFor_FK BIGINT NOT NULL,
    EnderecoPrimario BOOLEAN NOT NULL DEFAULT FALSE,
    Logradouro VARCHAR(255) NOT NULL,
    Numero VARCHAR(255) NOT NULL,
    Complemento VARCHAR(255) DEFAULT NULL,
    Bairro VARCHAR(255) NOT NULL,
    Cidade VARCHAR(255) NOT NULL,
    UF VARCHAR(2) NOT NULL,
    CEP VARCHAR(9) NOT NULL,
    PRIMARY KEY (id_Endereco),
    FOREIGN KEY (id_CliFor_FK) REFERENCES ClienteFornecedor(id_CliFor)
);


CREATE TABLE Contato (
    id_Contato BIGINT GENERATED ALWAYS AS IDENTITY,
    id_CliFor_FK BIGINT NOT NULL,
    TipoContato VARCHAR(255) NOT NULL,
    Info_do_Contato VARCHAR(255) NOT NULL,
    Contato_Principal BOOLEAN NOT NULL DEFAULT FALSE,
    PRIMARY KEY (id_Contato),
    FOREIGN KEY (id_CliFor_FK) REFERENCES ClienteFornecedor(id_CliFor)
);


CREATE TABLE Login (
    id_Login BIGINT GENERATED ALWAYS AS IDENTITY,
    Data_Login TIMESTAMP NOT NULL DEFAULT NOW(),
    Data_Logout TIMESTAMP DEFAULT NULL,
    id_usuario_FK BIGINT NOT NULL,
    Dispositivo_Logado VARCHAR(255) NOT NULL,
    Localizacao VARCHAR(255) NOT NULL,
    Navegador VARCHAR(255) NOT NULL,
    PRIMARY KEY (id_Login),
    FOREIGN KEY (id_usuario_FK) REFERENCES Usuario(id_usuario)
);

-- Login
CREATE INDEX idx_id_Login ON Login(Data_Login);
CREATE INDEX idx_id_usuario_FK ON Login(id_usuario_FK);


-- Fluxo de estados (derivado, NAO existe coluna de status):
--   Aberto     = data_efetivacao IS NULL
--   Em analise = data_efetivacao IS NOT NULL AND data_aprovacao IS NULL
--   Pago       = data_aprovacao IS NOT NULL
-- Quem manda no estado e data_efetivacao (carimbo do servidor), NAO data_pagamento:
-- data_pagamento e digitada no formulario e pode ser retroativa, entao um admin
-- editando a data de pagamento mudaria o estado do lancamento sem querer.
-- Dinheiro so entra no caixa (resumo/dashboard) e inadimplencia so limpa na APROVACAO.
CREATE TABLE Lancamento (
    id_lancamento BIGINT GENERATED ALWAYS AS IDENTITY,
    id_Usuario_FK_Lancamento BIGINT NOT NULL,
    id_CliFor_Relacionado_FK BIGINT NOT NULL,
    id_tipo_conta_FK BIGINT NOT NULL,
    -- UTC, igual aos carimbos de efetivacao/aprovacao/edicao (que vem do Python).
    -- NOW() puro devolveria a hora LOCAL do Postgres (o banco roda em America/Sao_Paulo),
    -- e a linha do tempo da tela, que ordena os quatro juntos, misturaria dois relogios.
    -- Quem converte para o fuso do usuario e o frontend.
    Data_lancamento TIMESTAMP NOT NULL DEFAULT timezone('utc', now()),
    Valor DECIMAL(15,2) NOT NULL,
    Data_vencimento DATE NOT NULL,
    Multa DECIMAL(15,2),
    Juros DECIMAL(15,2),
    -- Efetivacao (Operador ou Admin): manda o lancamento para "Em analise".
    id_Usuario_FK_Efetivacao BIGINT,          -- quem efetivou; vem do token, nunca do body
    data_efetivacao TIMESTAMP,                -- carimbo do servidor no ato da efetivacao
    -- Aprovacao (so Admin): tira de "Em analise" e leva a "Pago".
    -- ATENCAO: o nome "Fechamento" e historico e mente um pouco -- esta coluna e o APROVADOR.
    -- Nao foi renomeada para nao somar risco ao deploy que ja carrega o backfill.
    id_Usuario_FK_Fechamento BIGINT,          -- quem aprovou; vem do token, nunca do body
    data_aprovacao TIMESTAMP,                 -- carimbo do servidor no ato da aprovacao
    -- Edicao (so Admin, via PATCH /lancamento/{id}/editar).
    -- Guarda so a ULTIMA edicao: uma nova sobrescreve a anterior. A linha do tempo da tela
    -- mostra QUEM editou e QUANDO, nunca o que mudou -- por isso um par de colunas basta.
    id_Usuario_FK_Edicao BIGINT,              -- quem editou; vem do token, nunca do body
    data_edicao TIMESTAMP,                    -- carimbo do servidor no ato da edicao
    data_pagamento TIMESTAMP,                 -- data ECONOMICA do pagamento (digitada, pode ser retroativa)
    Valor_Pago DECIMAL(15,2),
    Observacao TEXT,
    Observacao_Pagamento TEXT,
    Natureza_Lancamento natureza_enum NOT NULL,
    Estorno BOOLEAN NOT NULL DEFAULT FALSE,
    Lote BIGINT DEFAULT NULL,
    Comprovante BYTEA DEFAULT NULL,
    Comprovante_nome VARCHAR(255) DEFAULT NULL,
    PRIMARY KEY (id_lancamento),
    FOREIGN KEY (id_Usuario_FK_Lancamento) REFERENCES Usuario(id_usuario),
    FOREIGN KEY (id_CliFor_Relacionado_FK) REFERENCES ClienteFornecedor(id_CliFor),
    FOREIGN KEY (id_tipo_conta_FK) REFERENCES tipo_conta(id_tipo_conta),
    FOREIGN KEY (id_Usuario_FK_Efetivacao) REFERENCES Usuario(id_usuario),
    FOREIGN KEY (id_Usuario_FK_Fechamento) REFERENCES Usuario(id_usuario),
    FOREIGN KEY (id_Usuario_FK_Edicao) REFERENCES Usuario(id_usuario),
    -- Os dois CHECKs abaixo valem para uma base criada do zero por este script.
    -- NAO estao aplicados na base existente: a migracao de main._aplicar_migracoes()
    -- so faz o ALTER (um CHECK que estoura no startup derruba o deploy inteiro).
    -- Quem garante estas invariantes hoje e o backend; para aplica-las no banco,
    -- ver a secao [OPCIONAL] de comentarios/migracao_em_analise.sql.
    --
    -- Nao se aprova o que nao foi efetivado.
    CONSTRAINT ck_lancamento_aprovacao_exige_efetivacao
        CHECK (data_aprovacao IS NULL OR data_efetivacao IS NOT NULL),
    -- Nao se efetiva sem informar a data de pagamento.
    CONSTRAINT ck_lancamento_efetivacao_exige_pagamento
        CHECK (data_efetivacao IS NULL OR data_pagamento IS NOT NULL)
);

-- Lancamento
CREATE INDEX idx_lancamento_data_vencimento ON Lancamento(Data_vencimento);
CREATE INDEX idx_lancamento_data_lancamento ON Lancamento(Data_lancamento);
CREATE INDEX idx_lancamento_estorno ON Lancamento(Estorno);
CREATE INDEX idx_lancamento_clifor ON Lancamento(id_CliFor_Relacionado_FK);
CREATE INDEX idx_lancamento_data_pagamento ON Lancamento(data_pagamento);
CREATE INDEX idx_lancamento_natureza ON Lancamento(Natureza_Lancamento);
CREATE INDEX idx_lancamento_tipo_conta ON Lancamento(id_tipo_conta_fk);
-- Fila de aprovacao do admin (apenas_em_analise) e filtro apenas_abertos.
CREATE INDEX idx_lancamento_em_analise ON Lancamento(data_efetivacao, data_aprovacao);


-- Registro de Tokens idx_token_ativo_usuario
-- Para exigir o token no login e dificultar o acesso por histórico.
CREATE TABLE token_ativo (
    id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    jti VARCHAR(255) NOT NULL UNIQUE,
    id_usuario_fk BIGINT NOT NULL REFERENCES usuario(id_usuario),
    criado_em TIMESTAMP NOT NULL DEFAULT NOW(),
    exp TIMESTAMP NOT NULL
);

CREATE INDEX idx_token_ativo_jti ON token_ativo(jti);
CREATE INDEX idx_token_ativo_usuario ON token_ativo(id_usuario_fk);


-- ============================================================
-- TRIGGER: atualiza inadimplente em ClienteFornecedor
-- Dispara após INSERT, UPDATE ou DELETE em Lancamento
-- ============================================================

CREATE OR REPLACE FUNCTION atualizar_inadimplente()
RETURNS TRIGGER AS $$
DECLARE
    v_id_clifor BIGINT;
BEGIN
    IF TG_OP = 'DELETE' THEN
        v_id_clifor := OLD.id_CliFor_Relacionado_FK;
    ELSE
        v_id_clifor := NEW.id_CliFor_Relacionado_FK;
    END IF;

    UPDATE clientefornecedor
    SET inadimplente = EXISTS (
        SELECT 1
        FROM lancamento
        WHERE id_CliFor_Relacionado_FK = v_id_clifor
          AND natureza_lancamento = 'Credito'
          AND data_pagamento IS NULL
          AND estorno = FALSE
          AND data_vencimento < CURRENT_DATE
    )
    WHERE id_clifor = v_id_clifor;

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;



CREATE OR REPLACE TRIGGER trigger_inadimplente
AFTER INSERT OR UPDATE OR DELETE ON lancamento
FOR EACH ROW
EXECUTE FUNCTION atualizar_inadimplente();

-- ================================================
-- LOG DE ATIVIDADE — rastreio de mutações por sessão (LGPD / auditoria)
-- ================================================

CREATE TABLE log_atividade (
    id_log        BIGINT GENERATED ALWAYS AS IDENTITY,
    id_login_fk   BIGINT DEFAULT NULL,
    id_usuario_fk BIGINT DEFAULT NULL,
    timestamp     TIMESTAMP NOT NULL DEFAULT NOW(),
    metodo        VARCHAR(10) NOT NULL,
    endpoint      VARCHAR(255) NOT NULL,
    entidade      VARCHAR(100) DEFAULT NULL,
    id_entidade   BIGINT DEFAULT NULL,
    descricao     VARCHAR(255) DEFAULT NULL,
    status_code   INTEGER NOT NULL,
    PRIMARY KEY (id_log),
    FOREIGN KEY (id_login_fk)   REFERENCES login(id_login)       ON DELETE SET NULL,
    FOREIGN KEY (id_usuario_fk) REFERENCES usuario(id_usuario)   ON DELETE SET NULL
);

CREATE INDEX idx_log_atividade_login   ON log_atividade(id_login_fk);
CREATE INDEX idx_log_atividade_usuario ON log_atividade(id_usuario_fk);
CREATE INDEX idx_log_atividade_ts      ON log_atividade(timestamp);



-- ================================================
-- SENHA_TOKEN — token de uso unico para definir senha
-- (cadastro, reset e "esqueci a senha")
-- ================================================
-- Guardamos apenas o SHA-256 do token, nunca o valor em claro: um dump do banco
-- nao rende tokens utilizaveis. O token cru viaja so no fragment da URL (#token=),
-- e trocado pelo corpo do POST e consumido uma unica vez (usado_em).

CREATE TABLE senha_token (
    id            BIGINT GENERATED ALWAYS AS IDENTITY,
    id_usuario_fk BIGINT NOT NULL,
    token_hash    VARCHAR(64) NOT NULL UNIQUE,   -- SHA-256 em hex
    finalidade    VARCHAR(20) NOT NULL,          -- "cadastro" | "reset"
    criado_em     TIMESTAMP NOT NULL DEFAULT NOW(),
    expira_em     TIMESTAMP NOT NULL,
    usado_em      TIMESTAMP DEFAULT NULL,        -- NULL = ainda nao consumido
    PRIMARY KEY (id),
    FOREIGN KEY (id_usuario_fk) REFERENCES usuario(id_usuario) ON DELETE CASCADE
);

-- Lookup do token na validacao: e o unico caminho de leitura da tabela.
CREATE INDEX idx_senha_token_hash ON senha_token(token_hash);
